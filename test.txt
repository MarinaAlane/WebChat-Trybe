const reorderList = (nicknames) => {
  console.log('nick', nick);
  const newList = [];
  nicknames.forEach((nickname) => {
    if (nickname === nick) {
      newList.unshift(nickname);
    } else {
      newList.push(nickname);
    }
  });
  console.log('newList', newList);

  return newList;
};
